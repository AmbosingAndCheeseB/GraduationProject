from bs4 import BeautifulSoup
import requests

def get_html(url):
    _html = ""
    resp = requests.get(url)
    if resp.status_code == 200:
        _html = resp.text
    return _html




web_url = 'https://www.google.com/'

html = get_html(web_url)

soup = BeautifulSoup(html.text, 'html.parser')

print(soup)
